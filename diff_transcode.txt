+def test_transcode_resolve_public_id_missing_returns_404(monkeypatch):

+def test_transcode_parse_error_returns_400(transcode_env):
+    view = transcode_module.VideoTranscodeView()
+
+    class BrokenRequest:
+        query_params = {}
+        user = _create_user()
+
+        @property
+        def data(self):
